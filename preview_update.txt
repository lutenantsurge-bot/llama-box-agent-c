+float WhisperInput::rms(const std::vector<float> &audio) {
+    if (audio.empty()) return 0.0f;
+    float sum = 0.0f;
+    for (float v : audio) {
+        sum += v * v;
+    }
+    return std::sqrt(sum / audio.size());
